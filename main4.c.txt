#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    int price;
    char title[100];
} Book;

void total_price(Book books[], int count);
void list_books(Book books[], int count);

int main() {
    Book books[] = {
        {1, 1000, "All the Light We Cannot See"},
        {2, 300, "The 38 Letters from J.D. Rockefeller to his son"},
        {3, 1000, "The Ballad of Songbirds and Snakes"},
        {4, 550, "Killers of the Flower Moon"},
        {5, 870, "Elon Musk"},
        {6, 1344, "Milk and Honey 12-Month 2024"}
    };

    int count = sizeof(books) / sizeof(books[0]);
    total_price(books, count);
    list_books(books, count);
    return 0;
}

void total_price(Book books[], int count) {
    int total = 0;
    for (int i = 0; i < count; i++) {
        total += books[i].price;
    }
    printf("總價格為：%d\n", total);
}

void list_books(Book books[], int count) {
    printf("書籍資料：\n");
    for (int i = 0; i < count; i++) {
        printf("Id: %d, Price: %d, Title: %s\n", books[i].id, books[i].price, books[i].title);
    }
}
